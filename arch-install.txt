二.安装系统
1.设置镜像源：
redlector --country China --sort rate --protocol https --sava /etc/pacman.d/mirrorlist
pacman -Syy
2.磁盘分区：
lsblk //查看磁盘设备
分区：
a：boot分区 512M
b：swap  1/2 memory
c：/分区
gdisk /dev/diskname
n


+512M
ef00
n


+1/2 memory
8200
n




w
3.格式化磁盘
mkfs.vfat /dev/diskname1
mkfs.brtfs /dev/diskname3
mkswap /dev/diskname2
4.挂载磁盘
mount dev/diskname3 /mnt
mkdir -p /mnt/boot/efi && mount /dev/diskname1 /mnt/boot/efi
swapon /dev/diskname2
5.pacstrap /mnt linux linux-firmware linux-headers base base-devel vim neovim bash-completion
6.生成表文件
genfstab -U /mnt 
genfstab -U /mnt >> /mnt/etc/fstab
7.进入系统
arch-chroot /mnt

8.配置基本软件
pacman -Syy
pacman -S grub efibootmgr efivar networkmanager os-prober amd-ucode/intel-ucode
grub-install /dev/diskname
nvim /etc/default/grub
GRUB_GFXMODE=1920X1080
grub-mkconfig -o /boot/grub/grub.cfg
systemctl enable NetworkManager
passwd root
exit
umount /mnt/boot/efi
umount /mnt
reboot

三.进入新系统并配置
1.连接网络
有线网络：跳过
无线网络：nmtui
2.设置主机名
nvim /etc/hostname
input hostname
nvim /etc/hosts
127.0.0.1   localhost
::1         localhost
127.0.0.1   hostname.localdomain hostname
3.设置时区
timedatectl set-timezone Asia/Shanghai
timedatectl set-ntp true
4.添加用户
useradd --create-home username
passwd username
usermod -aG wheel,users,storage,power,lp username
export EDITOR=vim
visudo
/%wheel
uncomment #%wheel ALL=(ALL) ALL
reboot
